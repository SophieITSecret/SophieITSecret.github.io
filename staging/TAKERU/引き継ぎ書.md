## 10. 技術的な注意事項

### Safariのキャッシュ
Safariはキャッシュが強固。反映されない場合は、設定→Safari→履歴とWebサイトデータを消去。

### localStorageのキー
- `takeru-theme`：dark / light
- `takeru-font`：small / medium / large

### コード編集の鉄則
**部分的な差し替えは危険。必ずフルファイルで差し替えること。**  
Claude Codeを使う場合は指示すれば対応可。

### translate.googの制約
- HTMLページには概ね有効
- PDFには効かない（フラグ2で対応）
- X-Frame-Optionsでブロックされるサイトはフラグを`0`に変更

## 11. 音声ファイル生成方法（VOICEPEAK CLI）

### 環境
- **ツール：** VOICEPEAK（Windows版）
- **インストール先：** `C:\Program Files\VOICEPEAK\`
- **ナレーター名：** `Japanese Female 1`（英語名必須）

### ハマりポイント
1. ナレーター名は必ず英語
2. `-s`オプションを使う（`-t`はエラー）
3. 数字は読み仮名に変換
4. `time.sleep(3.0)`必須
5. 5本ごとに10秒休憩
6. TAKERUのCSVはヘッダー行あり、本文は4列目（index 4）

### 生成スクリプト（voicepeak_takeru.py）

```python
import os, subprocess, wave, time, re
import pandas as pd

VP_DIR = r"C:\Program Files\VOICEPEAK"
VP_EXE = "voicepeak.exe"
NARRATOR = "Japanese Female 1"
EMOTION = "happy=20,fun=10"
SPEED = "100"
PITCH = "50"
CSV_FILE = os.path.abspath("TAKERUcard.csv")
OUTPUT_DIR = os.path.abspath("voices_wav")
TEMP_DIR = os.path.abspath("temp_work")

for d in [OUTPUT_DIR, TEMP_DIR]:
    if not os.path.exists(d): os.makedirs(d)

def split_text(text, max_len=40):
    parts = re.split(r'(?<=。)', text)
    result = []
    for part in parts:
        if not part.strip(): continue
        if len(part) > max_len:
            sub_parts = re.split(r'(?<=、)', part)
            buf = ""
            for sp in sub_parts:
                if len(buf) + len(sp) > max_len and buf:
                    result.append(buf); buf = sp
                else:
                    buf += sp
            if buf: result.append(buf)
        else:
            result.append(part)
    return [r for r in result if r.strip()]

def combine_wav(file_list, output_path):
    data = []
    for f in file_list:
        with wave.open(f, 'rb') as w:
            data.append([w.getparams(), w.readframes(w.getnframes())])
    with wave.open(output_path, 'wb') as out:
        out.setparams(data[0][0])
        for d in data: out.writeframes(d[1])

def generate():
    df = pd.read_csv(CSV_FILE, encoding='utf-8-sig', header=None).iloc[1:]
    processed = 0
    for _, row in df.iterrows():
        talk_id = str(row[0]).strip()
        full_text = str(row[4]).strip()
        final_output = os.path.join(OUTPUT_DIR, f"{talk_id}.wav")
        if os.path.exists(final_output): continue
        sentences = split_text(full_text)
        chunk_files = []
        print(f"処理中: {talk_id} ({len(sentences)}分割)")
        success = True
        for i, chunk in enumerate(sentences):
            chunk_file = os.path.join(TEMP_DIR, f"chunk_{talk_id}_{i}.wav")
            cmd = [os.path.join(VP_DIR, VP_EXE), "-s", chunk,
                   "-n", NARRATOR, "-e", EMOTION,
                   "--speed", SPEED, "--pitch", PITCH, "-o", chunk_file]
            result = subprocess.run(cmd, cwd=VP_DIR, capture_output=True, text=True)
            time.sleep(3.0)
            if result.returncode != 0:
                print(f"エラー (ID:{talk_id}, chunk:{i}): {chunk}")
                success = False; break
            chunk_files.append(chunk_file)
        if not success:
            for f in chunk_files:
                if os.path.exists(f): os.remove(f)
            print(f"⚠️ {talk_id} スキップ")
            continue
        combine_wav(chunk_files, final_output)
        for f in chunk_files:
            if os.path.exists(f): os.remove(f)
        processed += 1
        if processed % 5 == 0:
            print(f"--- {processed}本完了。10秒休憩 ---")
            time.sleep(10)
    print("完了！")

if __name__ == "__main__":
    generate()
```

### WAV→MP3変換（wav_to_mp3.py）

```python
import os
from pydub import AudioSegment

INPUT_DIR = "voices_wav"
OUTPUT_DIR = "voices"

if not os.path.exists(OUTPUT_DIR):
    os.makedirs(OUTPUT_DIR)

files = sorted([f for f in os.listdir(INPUT_DIR) if f.endswith(".wav")])
for i, filename in enumerate(files):
    mp3_name = filename.replace(".wav", ".mp3")
    mp3_path = os.path.join(OUTPUT_DIR, mp3_name)
    if os.path.exists(mp3_path):
        print(f"[{i+1}] スキップ: {mp3_name}"); continue
    audio = AudioSegment.from_wav(os.path.join(INPUT_DIR, filename))
    audio = audio.set_channels(1)
    audio.export(mp3_path, format="mp3", bitrate="128k")
    print(f"[{i+1}] 変換完了: {mp3_name}")
print("完了！")
```
